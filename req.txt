scope.openFile = function(url) {
    var newWindow = window.open('', '_blank', 'noopener,noreferrer');

    // Ожидаем, пока окно не загрузится и не будет доступно
    var checkWindowLoaded = setInterval(function() {
        if (newWindow && newWindow.document && newWindow.document.readyState === 'complete') {
            clearInterval(checkWindowLoaded);
            
            // Устанавливаем заголовок с небольшой задержкой
            setTimeout(function() {
                if (newWindow) {
                    newWindow.document.title = "file142";
                }
            }, 100); // Задержка в 100 миллисекунд
        }
    }, 100); // Проверяем каждые 100 миллисекунд

    // Переходим по URL в новом окне
    newWindow.location.href = url;
};
