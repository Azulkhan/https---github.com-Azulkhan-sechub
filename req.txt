
<select ng-model="selectModel" ng-change="updateTextArea()">
    <option value="value1">Value 1</option>
    <option value="value2">Value 2</option>
    <option value="value3">Value 3</option>
</select>

<textarea ng-model="textAreaModel" ng-disabled="selectModel === 'value1'"></textarea>

$scope.selectModel = ''; // модель для <select>
$scope.textAreaModel = ''; // модель для <textarea>

$scope.updateTextArea = function() {
    // Обновляем значение в <textarea> при изменении <select>
    $scope.textAreaModel = $scope.selectModel;
};
