protected void Page_Load(object sender, EventArgs e)
{
    if (!IsPostBack)
    {
        // Получаем DataSet из вашего класса, который взаимодействует с базой данных
        DataSet ds = YourClassWithMethods.CSA_BLOCK_PARENT_FULL();

        // Используем DataSet для инициализации меню
        InitializeMenu(ds);
    }
}

private void InitializeMenu(DataSet ds)
{
    foreach (DataRow row in ds.Tables[0].Rows)
    {
        DevExpress.Web.MenuItem menuItem = new DevExpress.Web.MenuItem();
        menuItem.Text = row["YourColumnName"].ToString(); // Используйте имя колонки из вашего DataSet
        ASPxMenu1.Items.Add(menuItem);
    }
}
