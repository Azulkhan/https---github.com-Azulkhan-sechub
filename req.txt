WITH all_dates AS (
    SELECT generate_series(
        '2024-01-01'::date,  -- Замените на вашу начальную дату
        '2024-12-31'::date,  -- Замените на вашу конечную дату
        '1 day'::interval
    ) AS date
)
SELECT date
FROM (
    SELECT date, 
           ROW_NUMBER() OVER (ORDER BY date) AS rn
    FROM all_dates
    WHERE EXTRACT(DOW FROM date) BETWEEN 1 AND 5  -- Только рабочие дни (понедельник-пятница)
) AS numbered_days
WHERE (rn - 1) % 6 = 0;  -- Каждый шестой рабочий день
