Понял. В этом случае вы можете использовать сервис или фабрику, чтобы передавать данные между контроллерами на одной странице. Вот пример:

```javascript
angular.module('myApp', [])
  .service('tempDataService', function() {
    var tempData = '';

    return {
      setData: function(data) {
        tempData = data;
      },
      getData: function() {
        return tempData;
      }
    };
  });

angular.module('myApp')
  .controller('FirstController', function($scope, tempDataService) {
    $scope.saveData = function(data) {
      tempDataService.setData(data);
    };
  });

angular.module('myApp')
  .controller('SecondController', function($scope, tempDataService) {
    $scope.getData = function() {
      $scope.data = tempDataService.getData();
    };
  });
```

Здесь `tempDataService` используется для сохранения данных из одного контроллера и их получения в другом. 

Пример HTML:

```html
<div ng-controller="FirstController">
  <input type="text" ng-model="dataToSave">
  <button ng-click="saveData(dataToSave)">Сохранить</button>
</div>

<div ng-controller="SecondController">
  <button ng-click="getData()">Получить данные</button>
  <p>Данные: {{ data }}</p>
</div>
```

Теперь, когда вы вводите данные и нажимаете кнопку "Сохранить" в первом контроллере, они сохраняются в сервисе `tempDataService`. Когда вы нажимаете кнопку "Получить данные" во втором контроллере, сохраненные данные выводятся.
