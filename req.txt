local str = "your_string_here"
local found_non_alphanumeric = false

for i = 1, #str do
    local char = str:sub(i, i)
    if not char:match("[\%w%-_%.%~]") then
        found_non_alphanumeric = true
        break
    end
end

if found_non_alphanumeric then
    print("Найден символ, не входящий в набор [%w%-_%.%~]")
else
    print("В строке нет символов, не входящих в набор [%w%-_%.%~]")
end
