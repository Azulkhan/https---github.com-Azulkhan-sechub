// AngularJS контроллер для управления интерфейсом
app.controller('MainController', function($scope, NcaLayerService) {
    $scope.document = null;

    // Функция для загрузки документа
    $scope.uploadDocument = function(file) {
        // Ваш код для загрузки документа
        $scope.document = file;
    };

    // Функция для подписи загруженного документа
