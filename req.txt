protected void Page_Load(object sender, EventArgs e)
{
    if (!IsPostBack)
    {
        InitializeComboBox();
    }
}

private void InitializeComboBox()
{
    DataSet ds = GetMenuDataSet(); // Получаем DataSet
    DataTable dt = ds.Tables[0]; // Предполагаем, что данные находятся в первой таблице

    // Настраиваем ASPxComboBox
    ASPxComboBox1.ValueType = typeof(int); // Устанавливаем тип значения
    ASPxComboBox1.TextField = "NAME"; // Имя поля, которое будет отображаться
    ASPxComboBox1.ValueField = "ID"; // Имя поля со значением, которое будет использоваться

    ASPxComboBox1.DataSource = dt; // Устанавливаем источник данных
    ASPxComboBox1.DataBind(); // Привязываем данные
}
