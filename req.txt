WITH RECURSIVE working_days AS (
    SELECT 
        start_date AS date,
        1 AS day_count
    FROM (SELECT '2024-10-01'::date AS start_date) AS sub
    WHERE start_date <= CURRENT_DATE

    UNION ALL

    SELECT 
        date + INTERVAL '1 day',
        CASE WHEN EXTRACT(DOW FROM date + INTERVAL '1 day') IN (0, 6) THEN day_count 
             ELSE day_count + 1 END
    FROM working_days
    WHERE day_count < 6
)

SELECT date
FROM working_days
WHERE day_count = 6;
