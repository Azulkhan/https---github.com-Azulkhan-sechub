В AngularJS можно использовать `$httpProvider` для конфигурации CORS заголовков. Вот пример использования `Access-Control-Allow-Origin` в AngularJS:

```javascript
angular.module('myApp', [])
  .config(['$httpProvider', function($httpProvider) {
    // Разрешаем доступ с определенного домена
    $httpProvider.defaults.headers.common['Access-Control-Allow-Origin'] = 'https://example.com';
  }])
  .controller('MyController', ['$scope', '$http', function($scope, $http) {
    $scope.loadData = function() {
      $http.get('https://api.example.com/data')
        .then(function(response) {
          $scope.data = response.data;
        })
        .catch(function(error) {
          console.error('Произошла ошибка при загрузке данных:', error);
        });
    };
  }]);
```

В этом примере мы используем `$httpProvider` для добавления заголовка `Access-Control-Allow-Origin` с указанным значением `'https://example.com'`. Это означает, что любые запросы, отправленные из приложения AngularJS на этом домене, будут иметь доступ к ресурсам на сервере, который возвращает этот заголовок.

Затем в контроллере `MyController` мы отправляем GET запрос к серверу по адресу `'https://api.example.com/data'` и обрабатываем успешный ответ, сохраняя данные в `$scope.data`, или выводим сообщение об ошибке в случае неудачи.
