
Извините за путаницу. В AngularJS нет прямой возможности получить доступ к имени файла, выбранного через поле ввода типа файла, с помощью директивы `ng-model`. Вместо этого, вы можете использовать объект события изменения `change` для получения этой информации. Вот как это можно сделать:

```html
<input type="file" id="fileInput" ng-change="getFileDetails($event)" />

<script>
app.controller('MainController', function($scope) {
    $scope.uploadedFileName = null;
    
    $scope.getFileDetails = function(event) {
        var files = event.target.files;
        if (files.length > 0) {
            $scope.uploadedFileName = files[0].name;
        }
    };
});
</script>
```

Этот код реагирует на изменения в поле ввода типа файла и вызывает функцию `getFileDetails()`, передавая объект события изменения. В этой функции мы можем получить доступ к выбранным файлам через `event.target.files`. Каждый файл представлен объектом File, у которого есть свойство `name`, содержащее имя файла.
