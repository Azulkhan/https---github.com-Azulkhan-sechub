WITH RECURSIVE working_days AS (
    SELECT 
        '2024-10-01'::date AS date
    UNION ALL
    SELECT 
        date + INTERVAL '1 day'
    FROM 
        working_days
    WHERE 
        date < '2024-10-15'  -- Конечная дата
),
filtered_working_days AS (
    SELECT 
        date
    FROM 
        working_days
    WHERE 
        EXTRACT(DOW FROM date) BETWEEN 1 AND 5  -- Рабочие дни (Пн-Пт)
)
SELECT 
    date
FROM 
    filtered_working_days
WHERE 
    date > CURRENT_DATE  -- Начинаем искать после текущей даты
ORDER BY 
    date
OFFSET 5 LIMIT 1;  -- Пропускаем первые 5 рабочих дней и берем следующий
