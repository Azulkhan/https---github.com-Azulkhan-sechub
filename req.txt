    <DropDownWindowTemplate>
        <dx:ASPxTreeView ID="treeView" runat="server" ShowCheckBoxes="True" AllowSelectNode="True">
            <%-- Код для заполнения древовидного списка --%>
        </dx:ASPxTreeView>
    </DropDownWindowTemplate>
</dx:ASPxDropDownEdit>


protected void Page_Load(object sender, EventArgs e)
{
    if (!IsPostBack)
    {
        InitializeTreeView();
    }
}

private void InitializeTreeView()
{
    ASPxTreeView treeView = dropDownEdit.FindControl("treeView") as ASPxTreeView;
    DataSet ds = GetMenuDataSet(); // Получаем DataSet
    DataTable dt = ds.Tables[0];

    // Используем рекурсивный метод для заполнения древовидного списка
    PopulateTreeViewNodes(treeView.Nodes, dt, 0);
}

private void PopulateTreeViewNodes(TreeViewNodeCollection nodesCollection, DataTable dataTable, int parentID)
{
    foreach (DataRow dataRow in dataTable.Select($"BLOK_ID = {parentID}"))
    {
        TreeViewNode node = new TreeViewNode();
        node.Name = dataRow["ID"].ToString();
        node.Text = dataRow["NAME"].ToString();
        nodesCollection.Add(node);

        // Рекурсивно заполняем дочерние узлы
        PopulateTreeViewNodes(node.Nodes, dataTable, Convert.ToInt32(dataRow["ID"]));
    }
}
