Понял. Для этого вам нужно настроить прокси-сервер таким образом, чтобы он перенаправлял ваши запросы на целевой URL. Это можно сделать с помощью различных технологий, таких как Node.js с использованием Express.js или любой другой серверной технологии.

Вот пример настройки прокси-сервера с использованием Node.js и Express.js:

```javascript
const express = require('express');
const bodyParser = require('body-parser');
const request = require('request');

const app = express();
const port = 3000;

app.use(bodyParser.json());

app.post('/your-route', (req, res) => {
  // Получаем данные из запроса
  const requestData = req.body;

  // Адрес целевого URL
  const targetUrl = 'адрес_целевого_URL';

  // Опции для запроса к целевому URL
  const options = {
    url: targetUrl,
    method: 'POST',
    json: true,
    body: requestData
  };

  // Отправляем запрос к целевому URL через прокси
  request(options, (error, response, body) => {
    if (error) {
      console.error(error);
      res.status(500).send('Произошла ошибка');
    } else {
      // Отправляем ответ обратно клиенту
      res.status(response.statusCode).json(body);
    }
  });
});

app.listen(port, () => {
  console.log(`Прокси-сервер запущен на порту ${port}`);
});
```

Этот пример создает прокси-сервер, который принимает POST запросы на `/your-route`, извлекает данные из запроса и отправляет их на целевой URL через `request`. Вы можете настроить `/your-route` на любой URL вашего выбора.
