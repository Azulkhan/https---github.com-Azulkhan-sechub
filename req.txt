WITH RECURSIVE working_days AS (
    SELECT start_date AS date
    FROM (SELECT '2024-10-01'::date AS start_date, '2024-10-15'::date AS end_date) AS params
    WHERE start_date <= end_date

    UNION ALL

    SELECT date + INTERVAL '1 day'
    FROM working_days
    WHERE date + INTERVAL '1 day' <= '2024-10-15' -- до конечной даты
)

SELECT date
FROM working_days
WHERE EXTRACT(DOW FROM date) NOT IN (0, 6) -- исключаем выходные (воскресенье и субботу)
GROUP BY date
HAVING COUNT(*) = 6 -- только шестые рабочие дни
  AND date > CURRENT_DATE -- выбираем только дни после сегодняшней даты
ORDER BY date
LIMIT 1;
