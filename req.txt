Чтобы использовать прокси при отправке POST запроса с помощью JavaScript, вы можете использовать объект XMLHttpRequest вместе с методом open() и указать адрес прокси как адрес назначения запроса. Вот пример:

```javascript
var xhr = new XMLHttpRequest();
var url = "адрес_прокси";
xhr.open("POST", url, true);
xhr.setRequestHeader("Content-Type", "application/json");
xhr.onreadystatechange = function () {
    if (xhr.readyState === 4 && xhr.status === 200) {
        // Обработка успешного ответа
        console.log(xhr.responseText);
    }
};
var data = JSON.stringify({"ключ": "значение"});
xhr.send(data);
```

Замените "адрес_прокси" на адрес вашего прокси-сервера, а также замените {"ключ": "значение"} на ваш объект данных, который вы хотите отправить.
