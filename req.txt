
protected void Page_Load(object sender, EventArgs e)
{
    if (!IsPostBack)
    {
        // Предполагается, что у вас есть метод GetMenuData(), который возвращает данные для меню
        var menuData = GetMenuData();
        InitializeMenu(menuData);
    }
}

private void InitializeMenu(IEnumerable<MenuDataItem> menuData)
{
    foreach (var item in menuData)
    {
        DevExpress.Web.MenuItem menuItem = new DevExpress.Web.MenuItem();
        menuItem.Text = item.Name; // item.Name должен быть заменен на имя поля из ваших данных
        ASPxMenu1.Items.Add(menuItem);
    }
}

</dx:ASPxMenu>
