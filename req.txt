
Понял. Вы можете сохранить информацию о выбранном файле в переменной области видимости контроллера AngularJS и обращаться к ней из других функций контроллера. Вот пример:

```html
<input type="file" id="fileInput" ng-model="uploadedFile" style="display: none;" />
<button ng-click="openFileInput()">Выбрать файл</button>
<button ng-click="getFileName()">Получить название файла</button>

<script>
app.controller('MainController', function($scope) {
    $scope.uploadedFile = null;
    $scope.uploadedFileName = null;

    $scope.openFileInput = function() {
        document.getElementById('fileInput').click();
    };

    $scope.getFileName = function() {
        if ($scope.uploadedFile) {
            $scope.uploadedFileName = $scope.uploadedFile.name;
            console.log("Имя файла: ", $scope.uploadedFileName);
        } else {
            console.log("Файл не выбран");
        }
    };
});
</script>
```

Этот код позволяет выбрать файл, нажав на кнопку "Выбрать файл", и затем получить его имя, нажав на кнопку "Получить название файла".
