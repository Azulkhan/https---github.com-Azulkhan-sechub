
<select ng-model="selectedOption">
  <option value="value1">Option 1</option>
  <option value="value2">Option 2</option>
  <option value="value3">Option 3</option>
</select>

<textarea ng-model="textAreaModel"></textarea>

angular.module('myApp', [])
  .controller('myController', function($scope) {
    // Инициализация значения selectedOption
    $scope.selectedOption = 'value1';

    // Функция для обновления значения textarea при изменении выбранной опции
    $scope.updateTextArea = function() {
      $scope.textAreaModel = $scope.selectedOption;
    };

    // Вызываем функцию updateTextArea для установки значения textarea
    $scope.updateTextArea();
  });
