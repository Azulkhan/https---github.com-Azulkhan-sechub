
angular.module('myApp', [])
  .controller('MyController', function($scope) {
    $scope.shouldExecuteFirstFunction = true;
    
    $scope.function1 = function() {
      console.log("Функция 1 выполнена");
    };
    
    $scope.function2 = function() {
      console.log("Функция 2 выполнена");
    };
    
    $scope.executeFunction = function() {
      if ($scope.shouldExecuteFirstFunction) {
        $scope.function1();
        $scope.shouldExecuteFirstFunction = false;
      } else {
        $scope.function2();
      }
    };
  });
