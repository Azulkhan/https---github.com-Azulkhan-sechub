protected void Page_Load(object sender, EventArgs e)
{
    if (!IsPostBack)
    {
        DataSet ds = YourDataAccessClass.GetMenuData();
        PopulateMenuItems(ASPxMenu1.Items, ds.Tables[0], null);
    }
}

private void PopulateMenuItems(MenuItemCollection itemsCollection, DataTable menuTable, object parentId)
{
    foreach (DataRow row in menuTable.Rows)
    {
        if (row["Block_ID"] == DBNull.Value ? parentId == null : row["Block_ID"].Equals(parentId))
        {
            MenuItem menuItem = new MenuItem();
            menuItem.Text = row["Name"].ToString();
            menuItem.NavigateUrl = row["NavigateUrl"].ToString(); // Если у вас есть URL
            menuItem.Value = row["ID"].ToString(); // Используйте ID как значение элемента меню

            itemsCollection.Add(menuItem);

            // Рекурсивно добавляем подменю для текущего пункта меню
            PopulateMenuItems(menuItem.Items, menuTable, row["ID"]);
        }
    }
}
